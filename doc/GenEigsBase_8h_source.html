84  // Real Ritz values calculated from UpperHessenbergEigen have exact zero imaginary part
85  // Complex Ritz values have exact conjugate pairs
86  // So we use exact tests here
87  static bool is_complex(const Complex& v) { return v.imag() != Scalar(0); }
88  static bool is_conj(const Complex& v1, const Complex& v2) { return v1 == Eigen::numext::conj(v2); }

90  // Implicitly restarted Arnoldi factorization
91  void restart(Index k)
92  {
93  using std::norm;
94 
95  if(k >= m_ncv)
96  return;
97 
98  DoubleShiftQR<Scalar> decomp_ds(m_ncv);
99  UpperHessenbergQR<Scalar> decomp_hb(m_ncv);
100  Matrix Q = Matrix::Identity(m_ncv, m_ncv);
101 
102  for(Index i = k; i < m_ncv; i++)
103  {
104  if(is_complex(m_ritz_val[i]) && is_conj(m_ritz_val[i], m_ritz_val[i + 1]))
105  {
106  // H - mu * I = Q1 * R1
107  // H <- R1 * Q1 + mu * I = Q1' * H * Q1
108  // H - conj(mu) * I = Q2 * R2
109  // H <- R2 * Q2 + conj(mu) * I = Q2' * H * Q2
110  //
111  // (H - mu * I) * (H - conj(mu) * I) = Q1 * Q2 * R2 * R1 = Q * R
112  const Scalar s = Scalar(2) * m_ritz_val[i].real();
113  const Scalar t = norm(m_ritz_val[i]);
114 
115  decomp_ds.compute(m_fac.matrix_H(), s, t);
116 
117  // Q -> Q * Qi
118  decomp_ds.apply_YQ(Q);
119  // H -> Q'HQ
120  // Matrix Q = Matrix::Identity(m_ncv, m_ncv);
121  // decomp_ds.apply_YQ(Q);
122  // m_fac_H = Q.transpose() * m_fac_H * Q;
123  m_fac.compress_H(decomp_ds);
124 
125  i++;
126  } else {
127  // QR decomposition of H - mu * I, mu is real
128  decomp_hb.compute(m_fac.matrix_H(), m_ritz_val[i].real());
129 
130  // Q -> Q * Qi
131  decomp_hb.apply_YQ(Q);
132  // H -> Q'HQ = RQ + mu * I
133  m_fac.compress_H(decomp_hb);
134  }
135  }
136 
137  m_fac.compress_V(Q);
138  m_fac.factorize_from(k, m_ncv, m_nmatop);
139 
140  retrieve_ritzpair();
141  }
142 
143  // Calculates the number of converged Ritz values
144  Index num_converged(Scalar tol)
145  {
146  // thresh = tol * max(m_eps23, abs(theta)), theta for Ritz value
147  Array thresh = tol * m_ritz_val.head(m_nev).array().abs().max(m_eps23);
148  Array resid = m_ritz_est.head(m_nev).array().abs() * m_fac.f_norm();
149  // Converged "wanted" Ritz values
150  m_ritz_conv = (resid < thresh);
151 
152  return m_ritz_conv.cast<Index>().sum();
153  }
154 
155  // Returns the adjusted nev for restarting
156  Index nev_adjusted(Index nconv)
157  {
158  using std::abs;
159 
160  Index nev_new = m_nev;
161  for(Index i = m_nev; i < m_ncv; i++)
162  if(abs(m_ritz_est[i]) < m_near_0) nev_new++;
163 
164  // Adjust nev_new, according to dnaup2.f line 660~674 in ARPACK
165  nev_new += std::min(nconv, (m_ncv - nev_new) / 2);
166  if(nev_new == 1 && m_ncv >= 6)
167  nev_new = m_ncv / 2;
168  else if(nev_new == 1 && m_ncv > 3)
169  nev_new = 2;
170 
171  if(nev_new > m_ncv - 2)
172  nev_new = m_ncv - 2;
173 
174  // Increase nev by one if ritz_val[nev - 1] and
175  // ritz_val[nev] are conjugate pairs
176  if(is_complex(m_ritz_val[nev_new - 1]) &&
177  is_conj(m_ritz_val[nev_new - 1], m_ritz_val[nev_new]))
178  {
179  nev_new++;
180  }
181 
182  return nev_new;
183  }

185  // Retrieves and sorts Ritz values and Ritz vectors
186  void retrieve_ritzpair()
187  {
188  UpperHessenbergEigen<Scalar> decomp(m_fac.matrix_H());
189  const ComplexVector& evals = decomp.eigenvalues();
190  ComplexMatrix evecs = decomp.eigenvectors();
191 
192  SortEigenvalue<Complex, SelectionRule> sorting(evals.data(), evals.size());
193  std::vector<int> ind = sorting.index();
194 
195  // Copy the Ritz values and vectors to m_ritz_val and m_ritz_vec, respectively
196  for(Index i = 0; i < m_ncv; i++)
197  {
198  m_ritz_val[i] = evals[ind[i]];
199  m_ritz_est[i] = evecs(m_ncv - 1, ind[i]);
200  }
201  for(Index i = 0; i < m_nev; i++)
202  {
203  m_ritz_vec.col(i).noalias() = evecs.col(ind[i]);
204  }
205  }

207 protected:
208  // Sorts the first nev Ritz pairs in the specified order
209  // This is used to return the final results
210  virtual void sort_ritzpair(int sort_rule)
211  {
212  // First make sure that we have a valid index vector
213  SortEigenvalue<Complex, LARGEST_MAGN> sorting(m_ritz_val.data(), m_nev);
214  std::vector<int> ind = sorting.index();
215 
216  switch(sort_rule)
217  {
218  case LARGEST_MAGN:
219  break;
220  case LARGEST_REAL:
221  {
222  SortEigenvalue<Complex, LARGEST_REAL> sorting(m_ritz_val.data(), m_nev);
223  ind = sorting.index();
224  }
225  break;
226  case LARGEST_IMAG:
227  {
228  SortEigenvalue<Complex, LARGEST_IMAG> sorting(m_ritz_val.data(), m_nev);
229  ind = sorting.index();
230  }
231  break;
232  case SMALLEST_MAGN:
233  {
234  SortEigenvalue<Complex, SMALLEST_MAGN> sorting(m_ritz_val.data(), m_nev);
235  ind = sorting.index();
236  }
237  break;
238  case SMALLEST_REAL:
239  {
240  SortEigenvalue<Complex, SMALLEST_REAL> sorting(m_ritz_val.data(), m_nev);
241  ind = sorting.index();
242  }
243  break;
244  case SMALLEST_IMAG:
245  {
246  SortEigenvalue<Complex, SMALLEST_IMAG> sorting(m_ritz_val.data(), m_nev);
247  ind = sorting.index();
248  }
249  break;
250  default:
251  throw std::invalid_argument("unsupported sorting rule");
252  }
253 
254  ComplexVector new_ritz_val(m_ncv);
255  ComplexMatrix new_ritz_vec(m_ncv, m_nev);
256  BoolArray new_ritz_conv(m_nev);
257 
258  for(Index i = 0; i < m_nev; i++)
259  {
260  new_ritz_val[i] = m_ritz_val[ind[i]];
261  new_ritz_vec.col(i).noalias() = m_ritz_vec.col(ind[i]);
262  new_ritz_conv[i] = m_ritz_conv[ind[i]];
263  }
264 
265  m_ritz_val.swap(new_ritz_val);
266  m_ritz_vec.swap(new_ritz_vec);
267  m_ritz_conv.swap(new_ritz_conv);
268  }

364  Index compute(Index maxit = 1000, Scalar tol = 1e-10, int sort_rule = LARGEST_MAGN)
365  {
366  // The m-step Arnoldi factorization
367  m_fac.factorize_from(1, m_ncv, m_nmatop);
368  retrieve_ritzpair();
369  // Restarting
370  Index i, nconv = 0, nev_adj;
371  for(i = 0; i < maxit; i++)
372  {
373  nconv = num_converged(tol);
374  if(nconv >= m_nev)
375  break;
376 
377  nev_adj = nev_adjusted(nconv);
378  restart(nev_adj);
379  }
380  // Sorting results
381  sort_ritzpair(sort_rule);
382 
383  m_niter += i + 1;
384  m_info = (nconv >= m_nev) ? SUCCESSFUL : NOT_CONVERGING;
385 
386  return std::min(m_nev, nconv);
387  }

412  ComplexVector eigenvalues() const
413  {
414  const Index nconv = m_ritz_conv.cast<Index>().sum();
415  ComplexVector res(nconv);
416 
417  if(!nconv)
418  return res;

420  Index j = 0;
421  for(Index i = 0; i < m_nev; i++)
422  {
423  if(m_ritz_conv[i])

426  j++;

428  }

442  ComplexMatrix eigenvectors(Index nvec) const
443  {
444  const Index nconv = m_ritz_conv.cast<Index>().sum();
445  nvec = std::min(nvec, nconv);
446  ComplexMatrix res(m_n, nvec);
447 
448  if(!nvec)
449  return res;
450 
451  ComplexMatrix ritz_vec_conv(m_ncv, nvec);
452  Index j = 0;
453  for(Index i = 0; i < m_nev && j < nvec; i++)
454  {
455  if(m_ritz_conv[i])
456  {
457  ritz_vec_conv.col(j).noalias() = m_ritz_vec.col(i);
458  j++;
459  }
460  }
461 
462  res.noalias() = m_fac.matrix_V() * ritz_vec_conv;
463 
464  return res;
465  }
