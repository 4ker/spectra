46  Scalar sigmai;

48  // First transform back the ritz values, and then sort
49  void sort_ritzpair(int sort_rule)
50  {
51  using std::abs;
52  using std::sqrt;
53 
54  // The eigenvalus we get from the iteration is
55  // nu = 0.5 * (1 / (lambda - sigma)) + 1 / (lambda - conj(sigma)))
56  // So the eigenvalues of the original problem is
57  // 1 \pm sqrt(1 - 4 * nu^2 * sigmai^2)
58  // lambda = sigmar + -----------------------------------
59  // 2 * nu
60  // We need to pick up the correct root
61  // Let vi be the i-th eigenvector, then A * vi = lambdai * vi
62  // and inv(A - r * I) * vi = 1 / (lambdai - r) * vi
63  // where r is any shift value.
64  // We can use this identity to back-solve lambdai
65 
66  // Select a random shift value
67  SimpleRandom<Scalar> rng(0);
68  Scalar shiftr = rng.random() * sigmar + rng.random();
69  Scalar shifti = rng.random() * sigmai + rng.random();
70  this->m_op->set_shift(shiftr, shifti);
71 
72  // Calculate inv(A - r * I) * vi
73  ComplexArray v(this->m_n), OPv(this->m_n);
74  Scalar eps = Eigen::NumTraits<Scalar>::epsilon();
75  for(int i = 0; i < this->m_nev; i++)
76  {
77  v = this->m_fac_V * this->m_ritz_vec.col(i);
78  this->m_op->perform_op(v.data(), OPv.data());
79 
80  // Two roots computed from the quadratic equation
81  Complex nu = this->m_ritz_val[i];
82  Complex root_part1 = sigmar + Scalar(0.5) / nu;
83  Complex root_part2 = Scalar(0.5) * sqrt(Scalar(1) - Scalar(4) * sigmai * sigmai * (nu * nu)) / nu;
84  Complex root1 = root_part1 + root_part2;
85  Complex root2 = root_part1 - root_part2;
86 
87  // Root computed from the linear equation
88  // Technically we can directly use this root, but its precision is usually
89  // lower than the one computed from the quadratic equation
90  int loc;
91  OPv.cwiseAbs().maxCoeff(&loc);
92  Complex lambdai = v[loc] / OPv[loc] + Complex(shiftr, shifti);
93 
94  if(abs(root1 - lambdai) < abs(root2 - lambdai))
95  lambdai = root1;
96  else
97  lambdai = root2;
98  this->m_ritz_val[i] = lambdai;
99 
100  if(abs(Eigen::numext::imag(lambdai)) > eps)
101  {
102  this->m_ritz_val[i + 1] = Eigen::numext::conj(lambdai);
103  i++;
104  } else {
105  this->m_ritz_val[i] = Complex(Eigen::numext::real(lambdai), Scalar(0));
106  }
107  }
108 
109  GenEigsSolver<Scalar, SelectionRule, OpType>::sort_ritzpair(sort_rule);
110  }
111 public:
131  GenEigsComplexShiftSolver(OpType* op_, int nev_, int ncv_, Scalar sigmar_, Scalar sigmai_) :
132  GenEigsSolver<Scalar, SelectionRule, OpType>(op_, nev_, ncv_),
133  sigmar(sigmar_), sigmai(sigmai_)
134  {
135  this->m_op->set_shift(sigmar, sigmai);
136  }
137 };
