48  // First transform back the ritz values, and then sort
49  void sort_ritzpair(int sort_rule)
50  {
51  // The eigenvalus we get from the iteration is
52  // nu = 0.5 * (1 / (lambda - sigma)) + 1 / (lambda - conj(sigma)))
53  // So the eigenvalues of the original problem is
54  // 1 \pm sqrt(1 - 4 * nu^2 * sigmai^2)
55  // lambda = sigmar + -----------------------------------
56  // 2 * nu
57  // We need to pick up the correct root
58  // Let vi be the i-th eigenvector, then A * vi = lambdai * vi
59  // and inv(A - r * I) * vi = 1 / (lambdai - r) * vi
60  // where r is any real value.
61  // We can use this identity to back-solve lambdai
62 
63  // Select an arbitrary real shift value
64  Scalar r = sigmar + std::sin(sigmar);
65  this->m_op->set_shift(r, 0);
66 
67  // Calculate inv(A - r * I) * vi
68  ComplexArray v;
69  Array v_real, v_imag;
70  Array lhs_real(this->m_n), lhs_imag(this->m_n);
71  Scalar eps = std::pow(std::numeric_limits<Scalar>::epsilon(), Scalar(2.0) / 3);
72  for(int i = 0; i < this->m_nev; i++)
73  {
74  v = this->m_fac_V * this->m_ritz_vec.col(i);
75  v_real = v.real();
76  v_imag = v.imag();
77 
78  this->m_op->perform_op(v_real.data(), lhs_real.data());
79  this->m_op->perform_op(v_imag.data(), lhs_imag.data());
80 
81  Complex lambdai = Complex(v_real[0], v_imag[0]) / Complex(lhs_real[0], lhs_imag[0]) +
82  Complex(r, 0);
83  this->m_ritz_val[i] = lambdai;
84 
85  if(std::abs(lambdai.imag()) > eps)
86  {
87  this->m_ritz_val[i + 1] = std::conj(lambdai);
88  i++;
89  }
90  }
91 
92  GenEigsSolver<Scalar, SelectionRule, OpType>::sort_ritzpair(sort_rule);
93  }
