134  // Arnoldi factorization starting from step-k
135  void factorize_from(int from_k, int to_m, const Vector &fk)
136  {
137  if(to_m <= from_k) return;
138 
139  m_fac_f = fk;
140 
141  Vector w(m_n);
142  Scalar beta = m_fac_f.norm();
143  // Keep the upperleft k x k submatrix of H and set other elements to 0
144  m_fac_H.rightCols(m_ncv - from_k).setZero();
145  m_fac_H.block(from_k, 0, m_ncv - from_k, from_k).setZero();
146  for(int i = from_k; i <= to_m - 1; i++)
147  {
148  bool restart = false;
149  // If beta = 0, then the next V is not full rank
150  // We need to generate a new residual vector that is orthogonal
151  // to the current V, which we call a restart
152  if(beta < m_prec)
153  {
154  SimpleRandom<Scalar> rng(2 * i);
155  m_fac_f.noalias() = rng.random_vec(m_n);
156  // f <- f - V * V' * f, so that f is orthogonal to V
157  MapMat V(m_fac_V.data(), m_n, i); // The first i columns
158  Vector Vf = V.transpose() * m_fac_f;
159  m_fac_f.noalias() -= V * Vf;
160  // beta <- ||f||
161  beta = m_fac_f.norm();
162 
163  restart = true;
164  }
165 
166  // v <- f / ||f||
167  m_fac_V.col(i).noalias() = m_fac_f / beta; // The (i+1)-th column
168 
169  // Note that H[i+1, i] equals to the unrestarted beta
170  if(restart)
171  m_fac_H(i, i - 1) = 0.0;
172  else
173  m_fac_H(i, i - 1) = beta;
174 
175  // w <- A * v, v = m_fac_V.col(i)
176  m_op->perform_op(&m_fac_V(0, i), w.data());
177  m_nmatop++;
178 
179  // First i+1 columns of V
180  MapMat Vs(m_fac_V.data(), m_n, i + 1);
181  // h = m_fac_H(0:i, i)
182  MapVec h(&m_fac_H(0, i), i + 1);
183  // h <- V' * w
184  h.noalias() = Vs.transpose() * w;
185 
186  // f <- w - V * h
187  m_fac_f.noalias() = w - Vs * h;
188  beta = m_fac_f.norm();
189 
190  if(beta > 0.717 * h.norm())
191  continue;
192 
193  // f/||f|| is going to be the next column of V, so we need to test
194  // whether V' * (f/||f||) ~= 0
195  Vector Vf = Vs.transpose() * m_fac_f;
196  // If not, iteratively correct the residual
197  int count = 0;
198  while(count < 5 && Vf.cwiseAbs().maxCoeff() > m_prec * beta)
199  {
200  // f <- f - V * Vf
201  m_fac_f.noalias() -= Vs * Vf;
202  // h <- h + Vf
203  h.noalias() += Vf;
204  // beta <- ||f||
205  beta = m_fac_f.norm();
206 
207  Vf.noalias() = Vs.transpose() * m_fac_f;
208  count++;
209  }
210  }
211  }

223  // Implicitly restarted Arnoldi factorization
224  void restart(int k)
225  {
226  if(k >= m_ncv)
227  return;
228 
229  DoubleShiftQR<Scalar> decomp_ds(m_ncv);
230  UpperHessenbergQR<Scalar> decomp_hb;
231  Matrix Q = Matrix::Identity(m_ncv, m_ncv);
232 
233  for(int i = k; i < m_ncv; i++)
234  {
235  if(is_complex(m_ritz_val[i], m_prec) && is_conj(m_ritz_val[i], m_ritz_val[i + 1], m_prec))
236  {
237  // H - mu * I = Q1 * R1
238  // H <- R1 * Q1 + mu * I = Q1' * H * Q1
239  // H - conj(mu) * I = Q2 * R2
240  // H <- R2 * Q2 + conj(mu) * I = Q2' * H * Q2
241  //
242  // (H - mu * I) * (H - conj(mu) * I) = Q1 * Q2 * R2 * R1 = Q * R
243  Scalar s = 2 * m_ritz_val[i].real();
244  Scalar t = std::norm(m_ritz_val[i]);
245 
246  decomp_ds.compute(m_fac_H, s, t);
247 
248  // Q -> Q * Qi
249  decomp_ds.apply_YQ(Q);
250  // H -> Q'HQ
251  // Matrix Q = Matrix::Identity(m_ncv, m_ncv);
252  // decomp_ds.apply_YQ(Q);
253  // m_fac_H = Q.transpose() * m_fac_H * Q;
254  m_fac_H = decomp_ds.matrix_QtHQ();
255 
256  i++;
257  } else {
258  // QR decomposition of H - mu * I, mu is real
259  m_fac_H.diagonal().array() -= m_ritz_val[i].real();
260  decomp_hb.compute(m_fac_H);
261 
262  // Q -> Q * Qi
263  decomp_hb.apply_YQ(Q);
264  // H -> Q'HQ = RQ + mu * I
265  m_fac_H = decomp_hb.matrix_RQ();
266  m_fac_H.diagonal().array() += m_ritz_val[i].real();
267  }
268  }
269  // V -> VQ, only need to update the first k+1 columns
270  // Q has some elements being zero
271  // The first (ncv - k + i) elements of the i-th column of Q are non-zero
272  Matrix Vs(m_n, k + 1);
273  int nnz;
274  for(int i = 0; i < k; i++)
275  {
276  nnz = m_ncv - k + i + 1;
277  MapMat V(m_fac_V.data(), m_n, nnz);
278  MapVec q(&Q(0, i), nnz);
279  Vs.col(i).noalias() = V * q;
280  }
281  Vs.col(k).noalias() = m_fac_V * Q.col(k);
282  m_fac_V.leftCols(k + 1).noalias() = Vs;
283 
284  Vector fk = m_fac_f * Q(m_ncv - 1, k - 1) + m_fac_V.col(k) * m_fac_H(k, k - 1);
285  factorize_from(k, m_ncv, fk);
286  retrieve_ritzpair();
287  }

289  // Calculate the number of converged Ritz values
290  int num_converged(Scalar tol)
291  {
292  // thresh = tol * max(m_prec, abs(theta)), theta for ritz value
293  Array thresh = tol * m_ritz_val.head(m_nev).array().abs().max(m_prec);
294  Array resid = m_ritz_est.head(m_nev).array().abs() * m_fac_f.norm();
295  // Converged "wanted" ritz values
296  m_ritz_conv = (resid < thresh);
297 
298  return m_ritz_conv.cast<int>().sum();
299  }

301  // Return the adjusted nev for restarting
302  int nev_adjusted(int nconv)
303  {
304  int nev_new = m_nev;
305 
306  for(int i = m_nev; i < m_ncv; i++)
307  if(std::abs(m_ritz_est[i]) < m_prec) nev_new++;
308 
309  // Increase nev by one if m_ritz_val[nev - 1] and
310  // m_ritz_val[nev] are conjugate pairs
311  if(is_complex(m_ritz_val[nev_new - 1], m_prec) &&
312  is_conj(m_ritz_val[nev_new - 1], m_ritz_val[nev_new], m_prec))
313  {
314  nev_new++;
315  }
316  // Adjust nev_new again, according to dnaup2.f line 660~674 in ARPACK
317  nev_new += std::min(nconv, (m_ncv - nev_new) / 2);
318  if(nev_new == 1 && m_ncv >= 6)
319  nev_new = m_ncv / 2;
320  else if(nev_new == 1 && m_ncv > 3)
321  nev_new = 2;
322 
323  if(nev_new > m_ncv - 2)
324  nev_new = m_ncv - 2;
325 
326  // Examine conjugate pairs again
327  if(is_complex(m_ritz_val[nev_new - 1], m_prec) &&
328  is_conj(m_ritz_val[nev_new - 1], m_ritz_val[nev_new], m_prec))
329  {
330  nev_new++;
331  }
332 
333  return nev_new;
334  }

336  // Retrieve and sort ritz values and ritz vectors
337  void retrieve_ritzpair()
338  {
339  UpperHessenbergEigen<Scalar> decomp(m_fac_H);
340  ComplexVector evals = decomp.eigenvalues();
341  ComplexMatrix evecs = decomp.eigenvectors();
342 
343  SortEigenvalue<Complex, SelectionRule> sorting(evals.data(), evals.size());
344  std::vector<int> ind = sorting.index();
345 
346  // Copy the ritz values and vectors to m_ritz_val and m_ritz_vec, respectively
347  for(int i = 0; i < m_ncv; i++)
348  {
349  m_ritz_val[i] = evals[ind[i]];
350  m_ritz_est[i] = evecs(m_ncv - 1, ind[i]);
351  }
352  for(int i = 0; i < m_nev; i++)
353  {
354  m_ritz_vec.col(i) = evecs.col(ind[i]);
355  }
356  }

358 protected:
359  // Sort the first nev Ritz pairs in the specified order
360  // This is used to return the final results
361  virtual void sort_ritzpair(int sort_rule)
362  {
363  // First make sure that we have a valid index vector
364  SortEigenvalue<Complex, LARGEST_MAGN> sorting(m_ritz_val.data(), m_nev);
365  std::vector<int> ind = sorting.index();
366 
367  switch(sort_rule)
368  {
369  case LARGEST_MAGN:
370  break;
371  case LARGEST_REAL:
372  {
373  SortEigenvalue<Complex, LARGEST_REAL> sorting(m_ritz_val.data(), m_nev);
374  ind = sorting.index();
375  }
376  break;
377  case LARGEST_IMAG:
378  {
379  SortEigenvalue<Complex, LARGEST_IMAG> sorting(m_ritz_val.data(), m_nev);
380  ind = sorting.index();
381  }
382  break;
383  case SMALLEST_MAGN:
384  {
385  SortEigenvalue<Complex, SMALLEST_MAGN> sorting(m_ritz_val.data(), m_nev);
386  ind = sorting.index();
387  }
388  break;
389  case SMALLEST_REAL:
390  {
391  SortEigenvalue<Complex, SMALLEST_REAL> sorting(m_ritz_val.data(), m_nev);
392  ind = sorting.index();
393  }
394  break;
395  case SMALLEST_IMAG:
396  {
397  SortEigenvalue<Complex, SMALLEST_IMAG> sorting(m_ritz_val.data(), m_nev);
398  ind = sorting.index();
399  }
400  break;
401  default:
402  throw std::invalid_argument("unsupported sorting rule");
403  }
404 
405  ComplexVector new_ritz_val(m_ncv);
406  ComplexMatrix new_ritz_vec(m_ncv, m_nev);
407  BoolArray new_ritz_conv(m_nev);
408 
409  for(int i = 0; i < m_nev; i++)
410  {
411  new_ritz_val[i] = m_ritz_val[ind[i]];
412  new_ritz_vec.col(i) = m_ritz_vec.col(ind[i]);
413  new_ritz_conv[i] = m_ritz_conv[ind[i]];
414  }
415 
416  m_ritz_val.swap(new_ritz_val);
417  m_ritz_vec.swap(new_ritz_vec);
418  m_ritz_conv.swap(new_ritz_conv);
419  }

465  void init(const Scalar *init_resid)
466  {
467  // Reset all matrices/vectors to zero
468  m_fac_V.resize(m_n, m_ncv);
469  m_fac_H.resize(m_ncv, m_ncv);
470  m_fac_f.resize(m_n);
471  m_ritz_val.resize(m_ncv);
472  m_ritz_vec.resize(m_ncv, m_nev);
473  m_ritz_est.resize(m_ncv);
474  m_ritz_conv.resize(m_nev);
475 
476  m_fac_V.setZero();
477  m_fac_H.setZero();
478  m_fac_f.setZero();
479  m_ritz_val.setZero();
480  m_ritz_vec.setZero();
481  m_ritz_est.setZero();
482  m_ritz_conv.setZero();
483 
484  Vector v(m_n);
485  std::copy(init_resid, init_resid + m_n, v.data());
486  Scalar vnorm = v.norm();
487  if(vnorm < m_prec)
488  throw std::invalid_argument("initial residual vector cannot be zero");
489  v /= vnorm;
490 
491  Vector w(m_n);
492  m_op->perform_op(v.data(), w.data());
493  m_nmatop++;
494 
495  m_fac_H(0, 0) = v.dot(w);
496  m_fac_f = w - v * m_fac_H(0, 0);
497  m_fac_V.col(0) = v;
498  }

534  int compute(int maxit = 1000, Scalar tol = 1e-10, int sort_rule = LARGEST_MAGN)
535  {
536  // The m-step Arnoldi factorization
537  factorize_from(1, m_ncv, m_fac_f);
538  retrieve_ritzpair();
539  // Restarting
540  int i, nconv = 0, nev_adj;
541  for(i = 0; i < maxit; i++)
542  {
543  nconv = num_converged(tol);
544  if(nconv >= m_nev)
545  break;
546 
547  nev_adj = nev_adjusted(nconv);
548  restart(nev_adj);
549  }
550  // Sorting results
551  sort_ritzpair(sort_rule);
552 
553  m_niter += i + 1;
554  m_info = (nconv >= m_nev) ? SUCCESSFUL : NOT_CONVERGING;
555 
556  return std::min(m_nev, nconv);
557  }

612  ComplexMatrix eigenvectors(int nvec)
613  {
614  int nconv = m_ritz_conv.cast<int>().sum();
615  nvec = std::min(nvec, nconv);
616  ComplexMatrix res(m_n, nvec);
617 
618  if(!nvec)
619  return res;
620 
621  ComplexMatrix ritz_vec_conv(m_ncv, nvec);
622  int j = 0;
623  for(int i = 0; i < m_nev && j < nvec; i++)
624  {
625  if(m_ritz_conv[i])
626  {
627  ritz_vec_conv.col(j) = m_ritz_vec.col(i);
628  j++;
629  }
630  }
631 
632  res.noalias() = m_fac_V * ritz_vec_conv;
633 
634  return res;

739 template <typename Scalar = double,
740  int SelectionRule = LARGEST_MAGN,
741  typename OpType = DenseGenComplexShiftSolve<double> >
742 class GenEigsComplexShiftSolver: public GenEigsSolver<Scalar, SelectionRule, OpType>
743 {
744 private:
745  typedef Eigen::Array<Scalar, Eigen::Dynamic, 1> Array;
746  typedef std::complex<Scalar> Complex;
747  typedef Eigen::Array<Complex, Eigen::Dynamic, 1> ComplexArray;
748 
749  Scalar sigmar;
750  Scalar sigmai;
751 
752  // First transform back the ritz values, and then sort
753  void sort_ritzpair(int sort_rule)
754  {
755  // The eigenvalus we get from the iteration is
756  // nu = 0.5 * (1 / (lambda - sigma)) + 1 / (lambda - conj(sigma)))
757  // So the eigenvalues of the original problem is
758  // 1 \pm sqrt(1 - 4 * nu^2 * sigmai^2)
759  // lambda = sigmar + -----------------------------------
760  // 2 * nu
761  // We need to pick up the correct root
762  // Let vi be the i-th eigenvector, then A * vi = lambdai * vi
763  // and inv(A - r * I) * vi = 1 / (lambdai - r) * vi
764  // where r is any real value.
765  // We can use this identity to back-solve lambdai
766 
767  // Select an arbitrary real shift value
768  Scalar r = sigmar + std::sin(sigmar);
769  this->m_op->set_shift(r, 0);
770 
771  // Calculate inv(A - r * I) * vi
772  ComplexArray v;
773  Array v_real, v_imag;
774  Array lhs_real(this->m_n), lhs_imag(this->m_n);
775  Scalar eps = std::pow(std::numeric_limits<Scalar>::epsilon(), Scalar(2.0) / 3);
776  for(int i = 0; i < this->m_nev; i++)
777  {
778  v = this->m_fac_V * this->m_ritz_vec.col(i);
779  v_real = v.real();
780  v_imag = v.imag();
781 
782  this->m_op->perform_op(v_real.data(), lhs_real.data());
783  this->m_op->perform_op(v_imag.data(), lhs_imag.data());
784 
785  Complex lambdai = Complex(v_real[0], v_imag[0]) / Complex(lhs_real[0], lhs_imag[0]) +
786  Complex(r, 0);
787  this->m_ritz_val[i] = lambdai;
788 
789  if(std::abs(lambdai.imag()) > eps)
790  {
791  this->m_ritz_val[i + 1] = std::conj(lambdai);
792  i++;
793  }
794  }
795 
796  GenEigsSolver<Scalar, SelectionRule, OpType>::sort_ritzpair(sort_rule);
797  }
798 public:

819  GenEigsSolver<Scalar, SelectionRule, OpType>(op_, nev_, ncv_),
820  sigmar(sigmar_), sigmai(sigmai_)
821  {
822  this->m_op->set_shift(sigmar, sigmai);
823  }
