181  // Arnoldi factorization starting from step-k
182  void factorize_from(int from_k, int to_m, const Vector& fk)
183  {
184  if(to_m <= from_k) return;
185 
186  m_fac_f = fk;
187 
188  Vector w(m_n);
189  Scalar beta = m_fac_f.norm();
190  // Keep the upperleft k x k submatrix of H and set other elements to 0
191  m_fac_H.rightCols(m_ncv - from_k).setZero();
192  m_fac_H.block(from_k, 0, m_ncv - from_k, from_k).setZero();
193  for(int i = from_k; i <= to_m - 1; i++)
194  {
195  bool restart = false;
196  // If beta = 0, then the next V is not full rank
197  // We need to generate a new residual vector that is orthogonal
198  // to the current V, which we call a restart
199  if(beta < m_prec)
200  {
201  SimpleRandom<Scalar> rng(2 * i);
202  m_fac_f.noalias() = rng.random_vec(m_n);
203  // f <- f - V * V' * f, so that f is orthogonal to V
204  MapMat V(m_fac_V.data(), m_n, i); // The first i columns
205  Vector Vf = V.transpose() * m_fac_f;
206  m_fac_f.noalias() -= V * Vf;
207  // beta <- ||f||
208  beta = m_fac_f.norm();
209 
210  restart = true;
211  }
212 
213  // v <- f / ||f||
214  m_fac_V.col(i).noalias() = m_fac_f / beta; // The (i+1)-th column
215 
216  // Note that H[i+1, i] equals to the unrestarted beta
217  if(restart)
218  m_fac_H(i, i - 1) = 0.0;
219  else
220  m_fac_H(i, i - 1) = beta;
221 
222  // w <- A * v, v = m_fac_V.col(i)
223  m_op->perform_op(&m_fac_V(0, i), w.data());
224  m_nmatop++;
225 
226  // First i+1 columns of V
227  MapMat Vs(m_fac_V.data(), m_n, i + 1);
228  // h = m_fac_H(0:i, i)
229  MapVec h(&m_fac_H(0, i), i + 1);
230  // h <- V' * w
231  h.noalias() = Vs.transpose() * w;
232 
233  // f <- w - V * h
234  m_fac_f.noalias() = w - Vs * h;
235  beta = m_fac_f.norm();
236 
237  if(beta > 0.717 * h.norm())
238  continue;
239 
240  // f/||f|| is going to be the next column of V, so we need to test
241  // whether V' * (f/||f||) ~= 0
242  Vector Vf = Vs.transpose() * m_fac_f;
243  // If not, iteratively correct the residual
244  int count = 0;
245  while(count < 5 && Vf.cwiseAbs().maxCoeff() > m_prec * beta)
246  {
247  // f <- f - V * Vf
248  m_fac_f.noalias() -= Vs * Vf;
249  // h <- h + Vf
250  h.noalias() += Vf;
251  // beta <- ||f||
252  beta = m_fac_f.norm();
253 
254  Vf.noalias() = Vs.transpose() * m_fac_f;
255  count++;
256  }
257  }
258  }

270  // Implicitly restarted Arnoldi factorization
271  void restart(int k)
272  {
273  if(k >= m_ncv)
274  return;
275 
276  DoubleShiftQR<Scalar> decomp_ds(m_ncv);
277  UpperHessenbergQR<Scalar> decomp_hb;
278  Matrix Q = Matrix::Identity(m_ncv, m_ncv);
279 
280  for(int i = k; i < m_ncv; i++)
281  {
282  if(is_complex(m_ritz_val[i], m_prec) && is_conj(m_ritz_val[i], m_ritz_val[i + 1], m_prec))
283  {
284  // H - mu * I = Q1 * R1
285  // H <- R1 * Q1 + mu * I = Q1' * H * Q1
286  // H - conj(mu) * I = Q2 * R2
287  // H <- R2 * Q2 + conj(mu) * I = Q2' * H * Q2
288  //
289  // (H - mu * I) * (H - conj(mu) * I) = Q1 * Q2 * R2 * R1 = Q * R
290  Scalar s = 2 * m_ritz_val[i].real();
291  Scalar t = std::norm(m_ritz_val[i]);
292 
293  decomp_ds.compute(m_fac_H, s, t);
294 
295  // Q -> Q * Qi
296  decomp_ds.apply_YQ(Q);
297  // H -> Q'HQ
298  // Matrix Q = Matrix::Identity(m_ncv, m_ncv);
299  // decomp_ds.apply_YQ(Q);
300  // m_fac_H = Q.transpose() * m_fac_H * Q;
301  m_fac_H = decomp_ds.matrix_QtHQ();
302 
303  i++;
304  } else {
305  // QR decomposition of H - mu * I, mu is real
306  m_fac_H.diagonal().array() -= m_ritz_val[i].real();
307  decomp_hb.compute(m_fac_H);
308 
309  // Q -> Q * Qi
310  decomp_hb.apply_YQ(Q);
311  // H -> Q'HQ = RQ + mu * I
312  m_fac_H = decomp_hb.matrix_RQ();
313  m_fac_H.diagonal().array() += m_ritz_val[i].real();
314  }
315  }
316  // V -> VQ, only need to update the first k+1 columns
317  // Q has some elements being zero
318  // The first (ncv - k + i) elements of the i-th column of Q are non-zero
319  Matrix Vs(m_n, k + 1);
320  int nnz;
321  for(int i = 0; i < k; i++)
322  {
323  nnz = m_ncv - k + i + 1;
324  MapMat V(m_fac_V.data(), m_n, nnz);
325  MapVec q(&Q(0, i), nnz);
326  Vs.col(i).noalias() = V * q;
327  }
328  Vs.col(k).noalias() = m_fac_V * Q.col(k);
329  m_fac_V.leftCols(k + 1).noalias() = Vs;
330 
331  Vector fk = m_fac_f * Q(m_ncv - 1, k - 1) + m_fac_V.col(k) * m_fac_H(k, k - 1);
332  factorize_from(k, m_ncv, fk);
333  retrieve_ritzpair();
334  }

336  // Calculate the number of converged Ritz values
337  int num_converged(Scalar tol)
338  {
339  // thresh = tol * max(m_prec, abs(theta)), theta for ritz value
340  Array thresh = tol * m_ritz_val.head(m_nev).array().abs().max(m_prec);
341  Array resid = m_ritz_est.head(m_nev).array().abs() * m_fac_f.norm();
342  // Converged "wanted" ritz values
343  m_ritz_conv = (resid < thresh);
344 
345  return m_ritz_conv.cast<int>().sum();
346  }

348  // Return the adjusted nev for restarting
349  int nev_adjusted(int nconv)
350  {
351  int nev_new = m_nev;
352 
353  for(int i = m_nev; i < m_ncv; i++)
354  if(std::abs(m_ritz_est[i]) < m_prec) nev_new++;
355 
356  // Increase nev by one if m_ritz_val[nev - 1] and
357  // m_ritz_val[nev] are conjugate pairs
358  if(is_complex(m_ritz_val[nev_new - 1], m_prec) &&
359  is_conj(m_ritz_val[nev_new - 1], m_ritz_val[nev_new], m_prec))
360  {
361  nev_new++;
362  }
363  // Adjust nev_new again, according to dnaup2.f line 660~674 in ARPACK
364  nev_new += std::min(nconv, (m_ncv - nev_new) / 2);
365  if(nev_new == 1 && m_ncv >= 6)
366  nev_new = m_ncv / 2;
367  else if(nev_new == 1 && m_ncv > 3)
368  nev_new = 2;
369 
370  if(nev_new > m_ncv - 2)
371  nev_new = m_ncv - 2;
372 
373  // Examine conjugate pairs again
374  if(is_complex(m_ritz_val[nev_new - 1], m_prec) &&
375  is_conj(m_ritz_val[nev_new - 1], m_ritz_val[nev_new], m_prec))
376  {
377  nev_new++;
378  }
379 
380  return nev_new;
381  }

383  // Retrieve and sort ritz values and ritz vectors
384  void retrieve_ritzpair()
385  {
386  UpperHessenbergEigen<Scalar> decomp(m_fac_H);
387  ComplexVector evals = decomp.eigenvalues();
388  ComplexMatrix evecs = decomp.eigenvectors();
389 
390  SortEigenvalue<Complex, SelectionRule> sorting(evals.data(), evals.size());
391  std::vector<int> ind = sorting.index();
392 
393  // Copy the ritz values and vectors to m_ritz_val and m_ritz_vec, respectively
394  for(int i = 0; i < m_ncv; i++)
395  {
396  m_ritz_val[i] = evals[ind[i]];
397  m_ritz_est[i] = evecs(m_ncv - 1, ind[i]);
398  }
399  for(int i = 0; i < m_nev; i++)
400  {
401  m_ritz_vec.col(i) = evecs.col(ind[i]);
402  }
403  }

405 protected:
406  // Sort the first nev Ritz pairs in the specified order
407  // This is used to return the final results
408  virtual void sort_ritzpair(int sort_rule)
409  {
410  // First make sure that we have a valid index vector
411  SortEigenvalue<Complex, LARGEST_MAGN> sorting(m_ritz_val.data(), m_nev);
412  std::vector<int> ind = sorting.index();
413 
414  switch(sort_rule)
415  {
416  case LARGEST_MAGN:
417  break;
418  case LARGEST_REAL:
419  {
420  SortEigenvalue<Complex, LARGEST_REAL> sorting(m_ritz_val.data(), m_nev);
421  ind = sorting.index();
422  }
423  break;
424  case LARGEST_IMAG:
425  {
426  SortEigenvalue<Complex, LARGEST_IMAG> sorting(m_ritz_val.data(), m_nev);
427  ind = sorting.index();
428  }
429  break;
430  case SMALLEST_MAGN:
431  {
432  SortEigenvalue<Complex, SMALLEST_MAGN> sorting(m_ritz_val.data(), m_nev);
433  ind = sorting.index();
434  }
435  break;
436  case SMALLEST_REAL:
437  {
438  SortEigenvalue<Complex, SMALLEST_REAL> sorting(m_ritz_val.data(), m_nev);
439  ind = sorting.index();
440  }
441  break;
442  case SMALLEST_IMAG:
443  {
444  SortEigenvalue<Complex, SMALLEST_IMAG> sorting(m_ritz_val.data(), m_nev);
445  ind = sorting.index();
446  }
447  break;
448  default:
449  throw std::invalid_argument("unsupported sorting rule");
450  }
451 
452  ComplexVector new_ritz_val(m_ncv);
453  ComplexMatrix new_ritz_vec(m_ncv, m_nev);
454  BoolArray new_ritz_conv(m_nev);
455 
456  for(int i = 0; i < m_nev; i++)
457  {
458  new_ritz_val[i] = m_ritz_val[ind[i]];
459  new_ritz_vec.col(i) = m_ritz_vec.col(ind[i]);
460  new_ritz_conv[i] = m_ritz_conv[ind[i]];
461  }
462 
463  m_ritz_val.swap(new_ritz_val);
464  m_ritz_vec.swap(new_ritz_vec);
465  m_ritz_conv.swap(new_ritz_conv);
466  }

512  void init(const Scalar* init_resid)
513  {
514  // Reset all matrices/vectors to zero
515  m_fac_V.resize(m_n, m_ncv);
516  m_fac_H.resize(m_ncv, m_ncv);
517  m_fac_f.resize(m_n);
518  m_ritz_val.resize(m_ncv);
519  m_ritz_vec.resize(m_ncv, m_nev);
520  m_ritz_est.resize(m_ncv);
521  m_ritz_conv.resize(m_nev);
522 
523  m_fac_V.setZero();
524  m_fac_H.setZero();
525  m_fac_f.setZero();
526  m_ritz_val.setZero();
527  m_ritz_vec.setZero();
528  m_ritz_est.setZero();
529  m_ritz_conv.setZero();
530 
531  Vector v(m_n);
532  std::copy(init_resid, init_resid + m_n, v.data());
533  Scalar vnorm = v.norm();
534  if(vnorm < m_prec)
535  throw std::invalid_argument("initial residual vector cannot be zero");
536  v /= vnorm;
537 
538  Vector w(m_n);
539  m_op->perform_op(v.data(), w.data());
540  m_nmatop++;
541 
542  m_fac_H(0, 0) = v.dot(w);
543  m_fac_f = w - v * m_fac_H(0, 0);
544  m_fac_V.col(0) = v;
545  }

581  int compute(int maxit = 1000, Scalar tol = 1e-10, int sort_rule = LARGEST_MAGN)
582  {
583  // The m-step Arnoldi factorization
584  factorize_from(1, m_ncv, m_fac_f);
585  retrieve_ritzpair();
586  // Restarting
587  int i, nconv = 0, nev_adj;
588  for(i = 0; i < maxit; i++)
589  {
590  nconv = num_converged(tol);
591  if(nconv >= m_nev)
592  break;
593 
594  nev_adj = nev_adjusted(nconv);
595  restart(nev_adj);
596  }
597  // Sorting results
598  sort_ritzpair(sort_rule);
599 
600  m_niter += i + 1;
601  m_info = (nconv >= m_nev) ? SUCCESSFUL : NOT_CONVERGING;
602 
603  return std::min(m_nev, nconv);
604  }
