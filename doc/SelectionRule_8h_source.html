95 // When comparing eigenvalues, we first calculate the "target"
96 // to sort. For example, if we want to choose the eigenvalues with
97 // largest magnitude, the target will be -std::abs(x).
98 // The minus sign is due to the fact that std::sort() sorts in ascending order.

100 // Default target: throw an exceptoin

222 // Sort eigenvalues and return the order index
223 template <typename PairType>
224 class PairComparator
225 {
226 public:
227  bool operator() (const PairType& v1, const PairType& v2)
228  {
229  return v1.first < v2.first;
230  }
231 };

233 template <typename T, int SelectionRule>
234 class SortEigenvalue
235 {
236 private:
237  typedef typename ElemType<T>::type TargetType; // Type of the sorting target, will be
238  // a floating number type, e.g. "double"
239  typedef std::pair<TargetType, int> PairType; // Type of the sorting pair, including
240  // the sorting target and the index
241 
242  std::vector<PairType> pair_sort;
243 
244 public:
245  SortEigenvalue(const T* start, int size) :
246  pair_sort(size)
247  {
248  for(int i = 0; i < size; i++)
249  {
250  pair_sort[i].first = SortingTarget<T, SelectionRule>::get(start[i]);
251  pair_sort[i].second = i;
252  }
253  PairComparator<PairType> comp;
254  std::sort(pair_sort.begin(), pair_sort.end(), comp);
255  }
256 
257  std::vector<int> index()
258  {
259  std::vector<int> ind(pair_sort.size());
260  for(unsigned int i = 0; i < ind.size(); i++)
261  ind[i] = pair_sort[i].second;
262 
263  return ind;
264  }
265 };
