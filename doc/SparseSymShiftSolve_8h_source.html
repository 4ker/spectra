67  void set_shift(Scalar sigma)
68  {
69  SparseMatrix mat = m_mat.template selfadjointView<Uplo>();
70  SparseMatrix identity(m_n, m_n);
71  identity.setIdentity();
72  mat = mat - sigma * identity;
73  m_solver.isSymmetric(true);
74  m_solver.compute(mat);
75  if(m_solver.info() != Eigen::Success)
76  throw std::invalid_argument("SparseSymShiftSolve: factorization failed with the given shift");
77  }
