83  // Implicitly restarted Lanczos factorization
84  void restart(Index k)
85  {
86  if(k >= m_ncv)
87  return;
88 
89  TridiagQR<Scalar> decomp(m_ncv);
90  Matrix Q = Matrix::Identity(m_ncv, m_ncv);
91 
92  for(Index i = k; i < m_ncv; i++)
93  {
94  // QR decomposition of H-mu*I, mu is the shift
95  decomp.compute(m_fac.matrix_H(), m_ritz_val[i]);
96 
97  // Q -> Q * Qi
98  decomp.apply_YQ(Q);
99  // H -> Q'HQ
100  // Since QR = H - mu * I, we have H = QR + mu * I
101  // and therefore Q'HQ = RQ + mu * I
102  m_fac.compress_H(decomp);
103  }
104 
105  m_fac.compress_V(Q);
106  m_fac.factorize_from(k, m_ncv, m_nmatop);
107 
108  retrieve_ritzpair();
109  }
110 
111  // Calculates the number of converged Ritz values
112  Index num_converged(Scalar tol)
113  {
114  // thresh = tol * max(m_eps23, abs(theta)), theta for Ritz value
115  Array thresh = tol * m_ritz_val.head(m_nev).array().abs().max(m_eps23);
116  Array resid = m_ritz_est.head(m_nev).array().abs() * m_fac.f_norm();
117  // Converged "wanted" Ritz values
118  m_ritz_conv = (resid < thresh);
119 
120  return m_ritz_conv.cast<Index>().sum();
121  }
122 
123  // Returns the adjusted nev for restarting
124  Index nev_adjusted(Index nconv)
125  {
126  using std::abs;
127 
128  Index nev_new = m_nev;
129  for(Index i = m_nev; i < m_ncv; i++)
130  if(abs(m_ritz_est[i]) < m_near_0) nev_new++;
131 
132  // Adjust nev_new, according to dsaup2.f line 677~684 in ARPACK
133  nev_new += std::min(nconv, (m_ncv - nev_new) / 2);
134  if(nev_new == 1 && m_ncv >= 6)
135  nev_new = m_ncv / 2;
136  else if(nev_new == 1 && m_ncv > 2)
137  nev_new = 2;
138 
139  if(nev_new > m_ncv - 1)
140  nev_new = m_ncv - 1;
141 
142  return nev_new;
143  }

145  // Retrieves and sorts Ritz values and Ritz vectors
146  void retrieve_ritzpair()
147  {
148  TridiagEigen<Scalar> decomp(m_fac.matrix_H());
149  const Vector& evals = decomp.eigenvalues();
150  const Matrix& evecs = decomp.eigenvectors();
151 
152  SortEigenvalue<Scalar, SelectionRule> sorting(evals.data(), evals.size());
153  std::vector<int> ind = sorting.index();
154 
155  // For BOTH_ENDS, the eigenvalues are sorted according
156  // to the LARGEST_ALGE rule, so we need to move those smallest
157  // values to the left
158  // The order would be
159  // Largest => Smallest => 2nd largest => 2nd smallest => ...
160  // We keep this order since the first k values will always be
161  // the wanted collection, no matter k is nev_updated (used in restart())
162  // or is nev (used in sort_ritzpair())
163  if(SelectionRule == BOTH_ENDS)
164  {
165  std::vector<int> ind_copy(ind);
166  for(Index i = 0; i < m_ncv; i++)
167  {
168  // If i is even, pick values from the left (large values)
169  // If i is odd, pick values from the right (small values)
170  if(i % 2 == 0)
171  ind[i] = ind_copy[i / 2];
172  else
173  ind[i] = ind_copy[m_ncv - 1 - i / 2];
174  }
175  }
176 
177  // Copy the Ritz values and vectors to m_ritz_val and m_ritz_vec, respectively
178  for(Index i = 0; i < m_ncv; i++)
179  {
180  m_ritz_val[i] = evals[ind[i]];
181  m_ritz_est[i] = evecs(m_ncv - 1, ind[i]);
182  }
183  for(Index i = 0; i < m_nev; i++)
184  {
185  m_ritz_vec.col(i).noalias() = evecs.col(ind[i]);
186  }
187  }

189 protected:
190  // Sorts the first nev Ritz pairs in the specified order
191  // This is used to return the final results
192  virtual void sort_ritzpair(int sort_rule)
193  {
194  // First make sure that we have a valid index vector
195  SortEigenvalue<Scalar, LARGEST_ALGE> sorting(m_ritz_val.data(), m_nev);
196  std::vector<int> ind = sorting.index();
197 
198  switch(sort_rule)
199  {
200  case LARGEST_ALGE:
201  break;
202  case LARGEST_MAGN:
203  {
204  SortEigenvalue<Scalar, LARGEST_MAGN> sorting(m_ritz_val.data(), m_nev);
205  ind = sorting.index();
206  }
207  break;
208  case SMALLEST_ALGE:
209  {
210  SortEigenvalue<Scalar, SMALLEST_ALGE> sorting(m_ritz_val.data(), m_nev);
211  ind = sorting.index();
212  }
213  break;
214  case SMALLEST_MAGN:
215  {
216  SortEigenvalue<Scalar, SMALLEST_MAGN> sorting(m_ritz_val.data(), m_nev);
217  ind = sorting.index();
218  }
219  break;
220  default:
221  throw std::invalid_argument("unsupported sorting rule");
222  }
223 
224  Vector new_ritz_val(m_ncv);
225  Matrix new_ritz_vec(m_ncv, m_nev);
226  BoolArray new_ritz_conv(m_nev);
227 
228  for(Index i = 0; i < m_nev; i++)
229  {
230  new_ritz_val[i] = m_ritz_val[ind[i]];
231  new_ritz_vec.col(i).noalias() = m_ritz_vec.col(ind[i]);
232  new_ritz_conv[i] = m_ritz_conv[ind[i]];
233  }
234 
235  m_ritz_val.swap(new_ritz_val);
236  m_ritz_vec.swap(new_ritz_vec);
237  m_ritz_conv.swap(new_ritz_conv);
238  }

332  Index compute(Index maxit = 1000, Scalar tol = 1e-10, int sort_rule = LARGEST_ALGE)
333  {
334  // The m-step Lanczos factorization
335  m_fac.factorize_from(1, m_ncv, m_nmatop);
336  retrieve_ritzpair();
337  // Restarting
338  Index i, nconv = 0, nev_adj;
339  for(i = 0; i < maxit; i++)
340  {
341  nconv = num_converged(tol);
342  if(nconv >= m_nev)
343  break;
344 
345  nev_adj = nev_adjusted(nconv);
346  restart(nev_adj);
347  }
348  // Sorting results
349  sort_ritzpair(sort_rule);
350 
351  m_niter += i + 1;
352  m_info = (nconv >= m_nev) ? SUCCESSFUL : NOT_CONVERGING;
353 
354  return std::min(m_nev, nconv);
355  }

410  virtual Matrix eigenvectors(Index nvec) const
411  {
412  const Index nconv = m_ritz_conv.cast<Index>().sum();
413  nvec = std::min(nvec, nconv);
414  Matrix res(m_n, nvec);
415 
416  if(!nvec)
417  return res;
418 
419  Matrix ritz_vec_conv(m_ncv, nvec);
420  Index j = 0;
421  for(Index i = 0; i < m_nev && j < nvec; i++)
422  {
423  if(m_ritz_conv[i])
424  {
425  ritz_vec_conv.col(j).noalias() = m_ritz_vec.col(i);
426  j++;
427  }
428  }
429 
430  res.noalias() = m_fac.matrix_V() * ritz_vec_conv;
431 
432  return res;
433  }
