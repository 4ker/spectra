193  // Arnoldi factorization starting from step-k
194  void factorize_from(int from_k, int to_m, const Vector &fk)
195  {
196  if(to_m <= from_k) return;
197 
198  fac_f = fk;
199 
200  Vector w(dim_n);
201  Scalar beta = 0.0, Hii = 0.0;
202  // Keep the upperleft k x k submatrix of H and set other elements to 0
203  fac_H.rightCols(ncv - from_k).setZero();
204  fac_H.block(from_k, 0, ncv - from_k, from_k).setZero();
205  for(int i = from_k; i <= to_m - 1; i++)
206  {
207  beta = fac_f.norm();
208  MapVec v(&fac_V(0, i), dim_n); // The (i+1)-th column
209  v.noalias() = fac_f / beta;
210  fac_H(i, i - 1) = beta;
211 
212  op->perform_op(v.data(), w.data());
213  nmatop++;
214 
215  Hii = v.dot(w);
216  fac_H(i - 1, i) = beta;
217  fac_H(i, i) = Hii;
218 
219  fac_f.noalias() = w - beta * fac_V.col(i - 1) - Hii * v;
220  // Correct f if it is not orthogonal to V
221  // Typically the largest absolute value occurs in
222  // the first element, i.e., <v1, f>, so we use this
223  // to test the orthogonality
224  Scalar v1f = fac_f.dot(fac_V.col(0));
225  if(v1f > prec || v1f < -prec)
226  {
227  Vector Vf(i + 1);
228  Vf.tail(i) = fac_V.block(0, 1, dim_n, i).transpose() * fac_f;
229  Vf[0] = v1f;
230  fac_f -= fac_V.leftCols(i + 1) * Vf;
231  }
232  }
233  }

235  // Implicitly restarted Arnoldi factorization
236  void restart(int k)
237  {
238  if(k >= ncv)
239  return;
240 
241  TridiagQR<Scalar> decomp;
242  Matrix Q = Matrix::Identity(ncv, ncv);
243 
244  for(int i = k; i < ncv; i++)
245  {
246  // QR decomposition of H-mu*I, mu is the shift
247  fac_H.diagonal().array() -= ritz_val[i];
248  decomp.compute(fac_H);
249 
250  // Q -> Q * Qi
251  decomp.apply_YQ(Q);
252  // H -> Q'HQ
253  // Since QR = H - mu * I, we have H = QR + mu * I
254  // and therefore Q'HQ = RQ + mu * I
255  fac_H = decomp.matrix_RQ();
256  fac_H.diagonal().array() += ritz_val[i];
257  }
258  // V -> VQ, only need to update the first k+1 columns
259  // Q has some elements being zero
260  // The first (ncv - k + i) elements of the i-th column of Q are non-zero
261  Matrix Vs(dim_n, k + 1);
262  int nnz;
263  for(int i = 0; i < k; i++)
264  {
265  nnz = ncv - k + i + 1;
266  MapMat V(fac_V.data(), dim_n, nnz);
267  MapVec q(&Q(0, i), nnz);
268  Vs.col(i).noalias() = V * q;
269  }
270  Vs.col(k).noalias() = fac_V * Q.col(k);
271  fac_V.leftCols(k + 1).noalias() = Vs;
272 
273  Vector fk = fac_f * Q(ncv - 1, k - 1) + fac_V.col(k) * fac_H(k, k - 1);
274  factorize_from(k, ncv, fk);
275  retrieve_ritzpair();
276  }

278  // Calculate the number of converged Ritz values
279  int num_converged(Scalar tol)
280  {
281  // thresh = tol * max(prec, abs(theta)), theta for ritz value
282  Array thresh = tol * ritz_val.head(nev).array().abs().max(prec);
283  Array resid = ritz_vec.template bottomRows<1>().transpose().array().abs() * fac_f.norm();
284  // Converged "wanted" ritz values
285  ritz_conv = (resid < thresh);
286 
287  return ritz_conv.cast<int>().sum();
288  }

290  // Return the adjusted nev for restarting
291  int nev_adjusted(int nconv)
292  {
293  int nev_new = nev;
294 
295  // Adjust nev_new, according to dsaup2.f line 677~684 in ARPACK
296  nev_new = nev + std::min(nconv, (ncv - nev) / 2);
297  if(nev == 1 && ncv >= 6)
298  nev_new = ncv / 2;
299  else if(nev == 1 && ncv > 2)
300  nev_new = 2;

305  // Retrieve and sort ritz values and ritz vectors
306  void retrieve_ritzpair()
307  {
308  TridiagEigen<double> decomp(fac_H);
309  Vector evals = decomp.eigenvalues();
310  Matrix evecs = decomp.eigenvectors();
311 
312  SortEigenvalue<Scalar, SelectionRule> sorting(evals.data(), evals.size());
313  std::vector<int> ind = sorting.index();
314 
315  // For BOTH_ENDS, the eigenvalues are sorted according
316  // to the LARGEST_ALGE rule, so we need to move those smallest
317  // values to the left
318  // The order would be
319  // Largest => Smallest => 2nd largest => 2nd smallest => ...
320  // We keep this order since the first k values will always be
321  // the wanted collection, no matter k is nev_updated (used in restart())
322  // or is nev (used in sort_ritzpair())
323  if(SelectionRule == BOTH_ENDS)
324  {
325  std::vector<int> ind_copy(ind);
326  for(int i = 0; i < ncv; i++)
327  {
328  // If i is even, pick values from the left (large values)
329  // If i is odd, pick values from the right (small values)
330  if(i % 2 == 0)
331  ind[i] = ind_copy[i / 2];
332  else
333  ind[i] = ind_copy[ncv - 1 - i / 2];
334  }
335  }
336 
337  // Copy the ritz values and vectors to ritz_val and ritz_vec, respectively
338  for(int i = 0; i < ncv; i++)
339  {
340  ritz_val[i] = evals[ind[i]];
341  }
342  for(int i = 0; i < nev; i++)
343  {
344  ritz_vec.col(i) = evecs.col(ind[i]);
345  }
346  }

349  // Sort the first nev Ritz pairs in the specified order
350  // This is used to return the final results
351  virtual void sort_ritzpair(int sort_rule)
352  {
353  // First make sure that we have a valid index vector
354  SortEigenvalue<Scalar, LARGEST_ALGE> sorting(ritz_val.data(), nev);
355  std::vector<int> ind = sorting.index();
356 
357  switch(sort_rule)
358  {
359  case LARGEST_ALGE:
360  break;
361  case LARGEST_MAGN:
362  {
363  SortEigenvalue<Scalar, LARGEST_MAGN> sorting(ritz_val.data(), nev);
364  ind = sorting.index();
365  }
366  break;
367  case SMALLEST_ALGE:
368  {
369  SortEigenvalue<Scalar, SMALLEST_ALGE> sorting(ritz_val.data(), nev);
370  ind = sorting.index();
371  }
372  break;
373  case SMALLEST_MAGN:
374  {
375  SortEigenvalue<Scalar, SMALLEST_MAGN> sorting(ritz_val.data(), nev);
376  ind = sorting.index();
377  }
378  break;
379  default:
380  throw std::invalid_argument("unsupported sorting rule");
381  }
382 
383  Vector new_ritz_val(ncv);
384  Matrix new_ritz_vec(ncv, nev);
385  BoolArray new_ritz_conv(nev);
386 
387  for(int i = 0; i < nev; i++)
388  {
389  new_ritz_val[i] = ritz_val[ind[i]];
390  new_ritz_vec.col(i) = ritz_vec.col(ind[i]);
391  new_ritz_conv[i] = ritz_conv[ind[i]];
392  }
393 
394  ritz_val.swap(new_ritz_val);
395  ritz_vec.swap(new_ritz_vec);
396  ritz_conv.swap(new_ritz_conv);
397  }

443  {
444  // Reset all matrices/vectors to zero
445  fac_V.resize(dim_n, ncv);
446  fac_H.resize(ncv, ncv);
447  fac_f.resize(dim_n);
448  ritz_val.resize(ncv);
449  ritz_vec.resize(ncv, nev);
450  ritz_conv.resize(nev);

452  fac_V.setZero();
453  fac_H.setZero();

456  ritz_vec.setZero();
457  ritz_conv.setZero();
458 
459  nmatop = 0;
460  niter = 0;
461 
462  Vector v(dim_n);
463  std::copy(init_resid, init_resid + dim_n, v.data());
464  Scalar vnorm = v.norm();
465  if(vnorm < prec)

471  nmatop++;

474  fac_f = w - v * fac_H(0, 0);
475  fac_V.col(0) = v;

510  int compute(int maxit = 1000, Scalar tol = 1e-10, int sort_rule = LARGEST_ALGE)
511  {
512  // The m-step Arnoldi factorization
513  factorize_from(1, ncv, fac_f);
514  retrieve_ritzpair();
515  // Restarting
516  int i, nconv = 0, nev_adj;
517  for(i = 0; i < maxit; i++)
518  {
519  nconv = num_converged(tol);
520  if(nconv >= nev)
521  break;
522 
523  nev_adj = nev_adjusted(nconv);
524  restart(nev_adj);
525  }
526  // Sorting results
527  sort_ritzpair(sort_rule);
528 
529  niter += i + 1;
530 
531  return std::min(nev, nconv);
532  }

581  Matrix eigenvectors(int nvec)
582  {
583  int nconv = ritz_conv.cast<int>().sum();
584  nvec = std::min(nvec, nconv);
585  Matrix res(dim_n, nvec);
586 
587  if(!nvec)
588  return res;
589 
590  Matrix ritz_vec_conv(ncv, nvec);
591  int j = 0;
592  for(int i = 0; i < nev && j < nvec; i++)
593  {
594  if(ritz_conv[i])
595  {
596  ritz_vec_conv.col(j) = ritz_vec.col(i);
597  j++;
598  }
599  }
600 
601  res.noalias() = fac_V * ritz_vec_conv;
602 
603  return res;
604  }
