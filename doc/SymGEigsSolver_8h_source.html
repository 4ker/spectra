298  // In generalized eigenvalue problem Ax=lambda*Bx, define the inner product to be <x, y> = x'By
299  Scalar inner_product(const Vector& x, const Vector& y)
300  {
301  m_Bop->mat_prod(y.data(), m_cache.data());
302  return x.dot(m_cache);
303  }
304  Scalar inner_product(const MapVec& x, const Vector& y)
305  {
306  m_Bop->mat_prod(y.data(), m_cache.data());
307  return x.dot(m_cache);
308  }
309  Vector inner_product(const MapMat& x, const Vector& y)
310  {
311  m_Bop->mat_prod(y.data(), m_cache.data());
312  return x.transpose() * m_cache;
313  }
314  // B-norm of a vector
315  Scalar norm(const Vector& x)
316  {
317  using std::sqrt;
318  return sqrt(inner_product(x, x));
319  }
